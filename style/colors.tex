%% style/colors.tex
%% The locked-in palette for Mathemagics. Five tcolorbox colors + supporting hues.
%% Tweak in ONE place; everything else inherits.

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Five tcolorbox colors — do not add new ones lightly. The discipline is
% deliberate: too many colored boxes makes a textbook visually noisy.
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

% Definitions: cool blue tint
\definecolor{mathemagicsDef}{HTML}{1B6EC2}        % blue rule
\definecolor{mathemagicsDefBg}{HTML}{EAF2FB}      % light blue tint

% Theorems / propositions / lemmas: minimal, white box, dark rule
\definecolor{mathemagicsThm}{HTML}{1A1A1A}        % near-black rule
\definecolor{mathemagicsThmBg}{HTML}{FFFFFF}      % white

% Intuition: warm cream
\definecolor{mathemagicsIntu}{HTML}{C09010}       % amber rule
\definecolor{mathemagicsIntuBg}{HTML}{FBF5E2}     % cream tint

% Example: pale gray
\definecolor{mathemagicsEx}{HTML}{555555}         % gray rule
\definecolor{mathemagicsExBg}{HTML}{F4F4F4}       % pale gray

% Aside (margin sidebar): no fill, just a thin orange accent
\definecolor{mathemagicsAside}{HTML}{C2521B}      % rust accent
\definecolor{mathemagicsAsideBg}{HTML}{FFFFFF}    % no tint (margin)

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Supporting colors for code, links, recap boxes, exercises
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\definecolor{mathemagicsLink}{HTML}{1B6EC2}       % matches definition blue
\definecolor{mathemagicsCodeBg}{HTML}{F7F7F7}     % code background
\definecolor{mathemagicsRecap}{HTML}{2E7D32}      % green for chapter recap
\definecolor{mathemagicsRecapBg}{HTML}{ECF5ED}    % green tint
\definecolor{mathemagicsWhereWeAre}{HTML}{6D4AA1} % purple for "where we are"
\definecolor{mathemagicsWhereWeAreBg}{HTML}{F1ECF7}
\definecolor{mathemagicsExercise}{HTML}{1A1A1A}   % black for exercise rule
\definecolor{mathemagicsColab}{HTML}{F9AB00}      % Colab badge yellow
\definecolor{mathemagicsCartoon}{HTML}{C2521B}    % matches aside (cartoon asides)
